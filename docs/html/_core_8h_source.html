86  bool OnEvent (const sf::Event& event) override
87  {
88  if (!Enabled()) return false;
89 
90  switch (event.type) {
91  case sf::Event::Closed: window->close(); return true;
92  case sf::Event::GainedFocus: hasFocus = true; return true;
93  case sf::Event::LostFocus: hasFocus = false; return true;
